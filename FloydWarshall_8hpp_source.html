101 template <class Graph>
102 void FloydWarshall<Graph>::compute(const Graph& graph)
103 {
104  idx_t i, j, k;
105  idx_t numNodes = graph.numNodes();
106  val_t weight;
107  distance_matrix_t dist (numNodes*numNodes);
108  idx_list_t next (numNodes*numNodes,GL_INF(idx_t));
109  // fill initial known edges
110  for (auto edge = graph.edge_cbegin(); edge != graph.edge_cend(); ++edge)
111  {
112  i = edge->source();
113  j = edge->dest();
114  weight = graph.getEdgeWeight(i,j);
115  // check for negative weights in undirected graphs
116  if (!graph.isDirected())
117  {
118  GL_ASSERT(weight >= 0,"FloydWarshall::compute | Graph is undirected and contains negative weights")
119  }
120  dist[i*numNodes+j].setDistance(weight);
121  next[i*numNodes+j] = j;
122  }
123  // set diagonals
124  for (idx_t i = 0; i < numNodes; ++i)
125  {
126  dist[i*numNodes+i].setDistance(0);
127  next[i*numNodes+i] = i;
128  }
129  // Actual Floyd-Warshall Algorithm
130  for (k = 0; k < numNodes; ++k) {
131  for (j = 0; j < numNodes; ++j) {
132  for (i = 0; i < numNodes; ++i) {
133  if (k == i || k == j)
134  continue;
135  if (dist[i*numNodes+k] + dist[k*numNodes+j] < dist[i*numNodes+j])
136  {
137  dist[i*numNodes+j] = dist[i*numNodes+k] + dist[k*numNodes+j];
138  next[i*numNodes+j] = next[i*numNodes+k];
139  }
140  }
141  }
142  }
143  // Check for negative cycles
144  for (idx_t i = 0; i < numNodes; ++i)
145  {
146  if (!dist[i*numNodes+i].isZero())
147  {
148  negativePath_ = {true,i};
149  }
150  }
151  dist_ = dist;
152  graph_ = graph;
153  next_ = next;
154  isInitialized_ = true;
155 }
